  <input type="email" id="email" placeholder="you@example.com" required />

  <button onclick="checkAttendance()">Search</button>

  <div id="result"></div>


  <script>

    const sheetURL = "https://script.google.com/macros/s/AKfycbyg7SMyyPChJtneFHbA1v5ET-BndbqVZglfy8ijevnkDoPrh9nRR3OTykwQi-Pd1Kl4/exec"; // Your deployed Apps Script URL


    async function checkAttendance() {

      const email = document.getElementById("email").value.trim().toLowerCase();

      const resultContainer = document.getElementById("result");

      resultContainer.innerHTML = "Loading...";


      try {

        const response = await fetch(sheetURL);

        const data = await response.json();


        const user = data.find(row => String(row.Email).toLowerCase() === email);


        if (!user) {

          resultContainer.innerHTML = `<p style="color:red;">No records found for this email.</p>`;

          return;

        }


        // Filter out only session keys (exclude Email, Total Attended, Attendance %)

        const sessionKeys = Object.keys(user).filter(

          key => key !== "Email" && !key.toLowerCase().includes("total") && !key.toLowerCase().includes("percent")

        );


        let presentCount = 0;

        let sessionData = `<table><tr><th>Session</th><th>Status</th></tr>`;


        sessionKeys.forEach(session => {

          const status = user[session] || "-";

          if (status.toUpperCase() === "P") presentCount++;

          sessionData += `<tr><td>${session}</td><td>${status}</td></tr>`;

        });


        sessionData += `</table>`;


        const total = user["Total Attended"] || presentCount;

        const percentage = user["Attendance %"] || (presentCount / sessionKeys.length);


        resultContainer.innerHTML = `

          <p><strong>Email:</strong> ${user.Email}</p>

          <p><strong>Total Sessions:</strong> ${sessionKeys.length}</p>

          <p><strong>Total Present (P):</strong> ${total}</p>

          <p><strong>Attendance Percentage:</strong> ${(percentage * 100).toFixed(2)}%</p>

          ${sessionData}

        `;

      } catch (err) {

        console.error("Fetch failed:", err);

        resultContainer.innerHTML = `<p style="color:red;">Error fetching attendance data: ${err.message}</p>`;

      }

    }

  </script>
